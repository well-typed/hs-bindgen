data Bar
instance Storable Bar
    where {sizeOf = \_ -> 16;
           alignment = \_ -> 8;
           peek = \x_0 -> (pure MkBar <*> peekByteOff x_0 0) <*> peekByteOff x_0 64;
           poke = \x_1 -> \x_2 -> case x_2 of
                                  {MkBar x_3
                                         y_4 -> pokeByteOff x_1 0 x_3 >> pokeByteOff x_1 64 y_4}}
