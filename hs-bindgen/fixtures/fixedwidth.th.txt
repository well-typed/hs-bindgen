data Foo
instance Storable Foo
    where {sizeOf = \_ -> 16;
           alignment = \_ -> 8;
           peek = \x_0 -> (pure MkFoo <*> peekByteOff x_0 0) <*> peekByteOff x_0 64;
           poke = \x_1 -> \x_2 -> case x_2 of
                                  {MkFoo sixty_four_3
                                         thirty_two_4 -> pokeByteOff x_1 0 sixty_four_3 >> pokeByteOff x_1 64 thirty_two_4}}
