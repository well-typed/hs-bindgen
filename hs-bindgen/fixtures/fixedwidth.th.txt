data CFoo
    = MkCFoo {cFoo_sixty_four :: CUint64T, cFoo_thirty_two :: CUint32T}
instance Storable CFoo
    where {sizeOf = \_ -> 16;
           alignment = \_ -> 8;
           peek = \x_0 -> (pure MkCFoo <*> peekByteOff x_0 0) <*> peekByteOff x_0 8;
           poke = \x_1 -> \x_2 -> case x_2 of
                                  {MkCFoo cFoo_sixty_four_3
                                          cFoo_thirty_two_4 -> pokeByteOff x_1 0 cFoo_sixty_four_3 >> pokeByteOff x_1 8 cFoo_thirty_two_4}}
