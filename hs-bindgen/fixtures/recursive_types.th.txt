instance Storable linked_list_a
    where {sizeOf = \_ -> 16;
           alignment = \_ -> 8;
           peek = \x_0 -> (pure Mklinked_list_a <*> peekByteOff x_0 0) <*> peekByteOff x_0 64;
           poke = \x_1 -> \x_2 -> case x_2 of
                                  {Mklinked_list_a head_3
                                                   tail_4 -> pokeByteOff x_1 0 head_3 >> pokeByteOff x_1 64 tail_4}}
