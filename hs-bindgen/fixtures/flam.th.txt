data CPascal = MkCPascal {cPascal_len :: CInt}
instance Storable CPascal
    where {sizeOf = \_ -> 4;
           alignment = \_ -> 4;
           peek = \ptr_0 -> pure MkCPascal <*> peekByteOff ptr_0 0;
           poke = \ptr_1 -> \s_2 -> case s_2 of
                                    {MkCPascal cPascal_len_3 -> pokeByteOff ptr_1 0 cPascal_len_3}}
